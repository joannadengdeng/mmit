Ex 22: MyLayerNorm

__init__: self.weight = nn.Parameter(torch.ones(dim))
          self.bias = nn.Parameter(torch.zeros(dim))
          self.eps = eps
forward:  mean = x.mean(-1, keepdim=True)
          var = x.var(-1, unbiased=False, keepdim=True)
          return self.weight * (x - mean) / torch.sqrt(var + self.eps) + self.bias

Ex 23: MyRMSNorm

__init__: self.weight = nn.Parameter(torch.ones(dim)); self.eps = eps
forward:  rms = torch.sqrt(x.pow(2).mean(-1, keepdim=True) + self.eps)
          return x / rms * self.weight

Ex 24: MyDropout

__init__: self.p = p

          return x * mask / (1 - self.p)

Ex 26: LoRALinear

          for p in base_linear.parameters(): p.requires_grad = False
          in_f, out_f = base_linear.in_features, base_linear.out_features
          self.A = nn.Parameter(torch.randn(rank, in_f) * 0.01)
          self.B = nn.Parameter(torch.zeros(out_f, rank))
forward:  return self.base(x) + self.scale * (x @ self.A.T @ self.B.T)

Ex 40: orthogonal_intervention

diff = W @ h.T - R @ h.T          # (D, B)
h_prime = h + (R.T @ diff).T      # (B, D)
I = torch.eye(R.shape[0], device=R.device)
ortho_penalty = ((R.T @ R - I) ** 2).sum()
