# 创建两个可训练参数: R (正交投影方向) 和 W (学习的引导)
self.R = nn.Parameter(torch.empty(rank, hidden_dim))   # (1, 4096)

这里用到的 PyTorch API

API做什么为什么用它
nn.Parameter(tensor)把 tensor 注册为可训练参数这样 optimizer 能找到它并更新它。普通 tensor 不会被训练
torch.empty(1, 4096)创建未初始化的 tensor下一行会用正交初始化覆盖，所以初始值无所谓
nn.init.orthogonal_(R)用正交矩阵填充 Rrank=1 时 R 变成单位向量。正交保证"投影→投回"不丢信息
nn.Linear(4096, 1)全连接层: y = Wx + b自动创建 weight (1,4096) 和 bias (1,) 两个 Parameter
nn.init.zeros_(tensor)全部填 0W=0 → 训练初期干预效果=0，模型行为不变，慢慢学
nn.Dropout(0.05)训练时随机 5% 值变 0防过拟合。推理时自动关闭
nn.Identity()什么都不做 (y = x)dropout=0 时的占位符，避免 if/else
关键设计：W 初始化为零 → 训练第一步时 Wh = 0，干预 delta ≈ 0，模型输出和原模型一样。然后 W 逐步学出非零值，模型被慢慢引导。这比一开始就大幅修改更安全。

Block 4: forward 路径 B — 选 top 1% by norm (Line 107-135)

LLaVA-1.5 实际走这条路（因为 vis_mask=None 但 steer_ratio=0.01 < 1.0）。从 620 个位置里选 norm 最大的 6 个，只修正它们。

# Step 1: 复制 h (不改原件)
h_out = h.clone()

# Step 2: 逐 batch 处理 (每个样本的选中位置可能不同)
for b in range(h.size(0)):
    vis_idx = torch.arange(h.size(1), device=h.device)  # [0,1,...,619] 全部候选

    # Step 3: 算每个位置的 norm, 选最大的 6 个
    norms = h[b, vis_idx].norm(dim=-1)       # (620,) 每个 token 的长度
    k = max(1, int(620 * 0.01))              # k = 6
    _, topk_local = norms.topk(k)            # 选 norm 最大的 6 个的位置
    vis_idx = vis_idx[topk_local]            # 最终 6 个绝对位置

    # Step 4: 对这 6 个位置做干预 (和路径 A 数学完全一样)
    h_sel = h[b, vis_idx]                    # (6, 4096)
    Rh = F.linear(h_sel, R)                  # (6, 1)
    Wh = self.W(h_sel)                       # (6, 1)
    delta = F.linear(self.dropout(Wh - Rh), R.t())  # (6, 4096)
    h_out[b, vis_idx] = h_sel + delta        # 写回 6 个位置

return h_out  # 614 个位置不变, 6 个被修正

Block 10: 注册 Hook — 最关键的工程代码 (Line 321-363)

# 对每个目标层注册 forward hook
for idx, iv in zip(target_indices, interventions):
    def make_hook(intervention):        # 闭包工厂 (避免循环变量捕获 bug)
        def hook(module, input, output):
            h = output[0] if isinstance(output, tuple) else output

            # 尝试构建 vis_mask (在 LLaVA 上会失败 → vis_mask=None)
            vis_mask = None
            # ... ids.shape[1] == h.shape[1] 检查 (45 ≠ 620 → 永远 False)

            h_new = intervention(h, vis_mask=vis_mask, steer_ratio=0.01)
            return (h_new,) + output[1:] if isinstance(output, tuple) else h_new
        return hook

    layer_list[idx].register_forward_hook(make_hook(iv))

# 再注册一个 pre-hook 捕获 input_ids
def _capture_input_ids(module, args, kwargs):
    module._mores_input_ids = kwargs.get("input_ids")

model.register_forward_pre_hook(_capture_input_ids, with_kwargs=True)
