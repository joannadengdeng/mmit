class Dog:
    species = "Canis"    # class attribute

    def __init__(self, name, age):
        self.name = name   # instance attr
        self.age = age

        return 0 < age < 30

    @property
    def human_years(self):
        return self.age * 7

import functools

def timer(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        import time
        start = time.time()
        result = func(*args, **kwargs)
        print(f"{func.__name__}: {time.time()-start}s")
        return result
    return wrapper

Decorator with Arguments
def retry(max_attempts=3):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for i in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception:
                    if i == max_attempts - 1:
                        raise
        return wrapper
    return decorator

@retry(max_attempts=5)
def flaky_call():
    ...
